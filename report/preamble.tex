% =====================================================================
%  preamble.tex — packages, layout, and custom commands
% =====================================================================

% --- Page layout ---
\usepackage[a4paper,margin=2.5cm]{geometry}
\usepackage{setspace}
\onehalfspacing

% --- Fonts / encoding ---
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}
\usepackage{lmodern}
\usepackage{microtype}            % subtle kerning/justification; fewer overfull lines

% --- Maths & units ---
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{siunitx}              
\sisetup{detect-all}

% --- Tables & figures ---
\usepackage{graphicx}
\graphicspath{{figures/}}
\usepackage{booktabs}
\usepackage{caption}
\usepackage{subcaption}
\usepackage{float}

% --- Bibliography (biblatex + biber) — Harvard / author-date ---
\usepackage{csquotes}             % recommended by biblatex; load BEFORE biblatex
\usepackage[backend=biber,
            style=authoryear,   
            sorting=nyt,         
            maxcitenames=3,      
            mincitenames=1,
            maxbibnames=99,      
            uniquename=false,
            uniquelist=false,
            dashed=false,        
            giveninits=true,     
            natbib=true]{biblatex}

% --- Cite Them Right / Imperial Harvard punctuation tweaks ---
\renewcommand*{\nameyeardelim}{\addcomma\space} % "Author, Year" in text
\DeclareNameAlias{sortname}{family-given}       % reference list: "Surname, F."
\DeclareFieldFormat[article]{title}{#1}         % article titles: no quotes
\DeclareFieldFormat[inproceedings]{title}{#1}
\DeclareFieldFormat{url}{Available at\addcolon\space\url{#1}} % "Available at: ..."
\DeclareFieldFormat{urldate}{(Accessed\addcolon\space#1)}     % "(Accessed: date)"

% --- Cross-referencing / hyperlinks ---
\usepackage{hyperref}
\usepackage{bookmark}             % clean PDF bookmarks in one pass (after hyperref)
\usepackage{cleveref}
\hypersetup{
    colorlinks=true,
    linkcolor=black,
    citecolor=black,
    urlcolor=blue,
}

% --- Headers/footers ---
% Match reference report: no running header, page number centred at the foot.
\pagestyle{plain}

% --- Compact chapter headings ---
\usepackage{titlesec}
% Chapter heading as "1  Introduction" (number + title on one line), matching
% the numbered subheadings 1.1, 1.1.1 — not the two-line "Chapter 1 / Title".
\titleformat{\chapter}[hang]
  {\normalfont\huge\bfseries}{\thechapter}{1em}{}
\titlespacing*{\chapter}{0pt}{0pt}{2.5ex}

% --- Code listings (if needed for pipeline snippets) ---
\usepackage{listings}
\usepackage{xcolor}
\lstset{
    basicstyle=\ttfamily\small,
    breaklines=true,
    frame=single,
    numbers=left,
    numberstyle=\tiny,
}
