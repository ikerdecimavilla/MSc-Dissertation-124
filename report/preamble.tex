% =====================================================================
%  preamble.tex — packages, layout, and custom commands
% =====================================================================

% --- Page layout ---
\usepackage[a4paper,margin=2.5cm]{geometry}
\usepackage{setspace}
\onehalfspacing

% --- Fonts / encoding ---
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}
\usepackage{lmodern}
\usepackage{microtype} % subtle kerning/justification; fewer overfull lines

% --- Comments ---
\usepackage{comment}

% --- Maths & units ---
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{siunitx}              
\sisetup{mode = match, propagate-math-font = true}

% --- Custom math commands ---
\newcommand{\lambdabar}{\bar{\lambda}}   % normalised (non-dimensional) slenderness

% --- Tables & figures ---
\usepackage{graphicx}
\graphicspath{{figures/}}
\usepackage{booktabs}
\usepackage{caption}
\usepackage{subcaption}
\usepackage{float}

% --- Bibliography (biblatex + biber) — Harvard / author-date ---
\usepackage{csquotes} % recommended by biblatex; load BEFORE biblatex
\usepackage[backend=biber,
            style=authoryear,   
            sorting=nyt,         
            maxcitenames=3,      
            mincitenames=1,
            maxbibnames=99,      
            uniquename=false,
            uniquelist=false,
            dashed=false,        
            giveninits=true,     
            natbib=true]{biblatex}

% --- Cite Them Right / Imperial Harvard punctuation tweaks ---
\renewcommand*{\nameyeardelim}{\addcomma\space} % "Author, Year" in text
\DeclareNameAlias{sortname}{family-given}       % reference list: "Surname, F."
\DeclareFieldFormat[article]{title}{#1}         % article titles: no quotes
\DeclareFieldFormat[inproceedings]{title}{#1}
\DeclareFieldFormat{url}{Available at\addcolon\space\url{#1}} % "Available at: ..."
\DeclareFieldFormat{urldate}{(Accessed\addcolon\space#1)}     % "(Accessed: date)"

% --- Compact chapter headings ---
\usepackage{titlesec}
\titleformat{\chapter}[hang]
  {\normalfont\huge\bfseries}{\thechapter}{1em}{}
\titlespacing*{\chapter}{0pt}{0pt}{2.5ex}

% --- Code listings ---
\usepackage{listings}
\usepackage{xcolor}
\lstset{
    basicstyle=\ttfamily\small,
    breaklines=true,
    frame=single,
    numbers=left,
    numberstyle=\tiny,
}

% --- Cross-referencing / hyperlinks (MUST BE LOADED AT THE VERY END) ---
\usepackage{hyperref}
\usepackage{bookmark} % clean PDF bookmarks in one pass
\usepackage{cleveref} % MUST be loaded after hyperref

\hypersetup{
    colorlinks=true,
    linkcolor=black,
    citecolor=black,
    urlcolor=blue,
}

% Reference names: always "Figure 2.1", "Table 2.1", "Eq. (2.1)"
\crefname{figure}{Figure}{Figures}
\Crefname{figure}{Figure}{Figures}
\crefname{table}{Table}{Tables}
\Crefname{table}{Table}{Tables}
\crefname{equation}{Eq.}{Eqs.}
\Crefname{equation}{Equation}{Equations}

% --- Headers/footers ---
\pagestyle{plain}
